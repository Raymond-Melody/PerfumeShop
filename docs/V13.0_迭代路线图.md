    container.innerHTML = '';

        var skeleton = document.createElement('div');
        skeleton.className = 'product-skeleton skeleton';

    var skeletons = document.querySelectorAll('.skeleton');

}

### V13.3: 图片CDN加速（优先级P1）

**目标**: 静态资源全球分发，提升海外用户访问速度

**功能清单**:
- [ ] 配置阿里云OSS/腾讯云COS
- [ ] 图片上传自动同步CDN
- [ ] 图片URL替换为CDN域名
- [ ] CDN缓存策略配置
- [ ] 图片压缩优化（WebP）

**技术实现**:
```asp
' 图片URL CDN化
Function GetImageCDNURL(localPath)
    Dim cdnDomain
    cdnDomain = "https://cdn.perfumeshop.com"
    
    If InStr(localPath, "http") = 1 Then
        GetImageCDNURL = localPath
    Else
        GetImageCDNURL = cdnDomain & localPath
    End If
End Function

' 使用示例
<img src="<%= GetImageCDNURL(rs("ImageURL")) %>">
```

**CDN配置**:
```xml
<!-- web.config CDN路由 -->
<rewrite>
    <rules>
        <rule name="Image CDN" stopProcessing="true">
            <match url="^images/(.*)" />
            <action type="Rewrite" url="https://cdn.perfumeshop.com/images/{R:1}" />
        </rule>
    </rules>
</rewrite>
```

**预期效果**:
- 海外用户加载时间: 3.5s → 1.2s (⬇️ 66%)
- 图片加载失败率: 5% → <1%
- 服务器带宽节省: 60-70%

### V13.4: Service Worker缓存（优先级P1）

**目标**: 实现离线浏览，提升二次访问速度

**功能清单**:
- [ ] 创建`sw.js`（Service Worker）
- [ ] 注册Service Worker
- [ ] 缓存静态资源
- [ ] 离线页面支持
- [ ] 后台同步

**技术实现**:
```javascript
// sw.js - Service Worker
var CACHE_NAME = 'perfumeshop-v13';
var CACHE_URLS = [
    '/',
    '/css/style.css',
    '/css/responsive.css',
    '/js/main.js',
    '/images/logo.png'
];

self.addEventListener('install', function(event) {
    event.waitUntil(
        caches.open(CACHE_NAME)
            .then(function(cache) {
                return cache.addAll(CACHE_URLS);
            })
    );
});

self.addEventListener('fetch', function(event) {
    event.respondWith(
        caches.match(event.request)
            .then(function(response) {
                return response || fetch(event.request);
            })
    );
});
```

```javascript
// 注册Service Worker
if ('serviceWorker' in navigator) {
    window.addEventListener('load', function() {
        navigator.serviceWorker.register('/sw.js')
            .then(function(registration) {
                console.log('SW注册成功:', registration.scope);
            })
            .catch(function(error) {
                console.log('SW注册失败:', error);
            });
    });
}
```

**预期效果**:
- 二次访问加载时间: 2.0s → 0.5s (⬇️ 75%)
- 离线页面可用: 80%核心页面
- 用户留存率提升: 20-25%

**目标**: DNS预解析+资源预取，进一步提升加载速度

**功能清单**:

- [ ] 资源预连接（preconnect）
- [ ] 关键资源预加载（preload）
- [ ] 空闲时预加载（prefetch）

**技术实现**:
```html
<head>
    <!-- DNS预解析 -->
    <link rel="dns-prefetch" href="//cdn.perfumeshop.com">
    <link rel="dns-prefetch" href="//fonts.googleapis.com">
    
    <!-- 预连接 -->
    <link rel="preconnect" href="https://cdn.perfumeshop.com" crossorigin>
    
    <!-- 关键资源预加载 -->
    <link rel="preload" href="/css/style.css" as="style">
    <link rel="preload" href="/js/main.js" as="script">
    
    <!-- 空闲时预加载 -->
    <link rel="prefetch" href="/products.asp">
    <link rel="prefetch" href="/cart.asp">
</head>
```

**预期效果**:
- DNS解析时间: 200ms → 50ms (⬇️ 75%)
- 首次渲染时间: 1.5s → 1.0s (️ 33%)
