@code {
    [Inject] private PerfumeShopContext Db { get; set; } = default!;
    private bool _loaded;
    private decimal _marketingExpense, _totalRevenue, _roi, _costRatio;
    private List<MarketingCampaign> _campaigns = new();

    protected override async Task OnInitializedAsync()
    {
        _campaigns = await Db.MarketingCampaigns.AsNoTracking().ToListAsync();
        _marketingExpense = _campaigns.Sum(c => c.ActualSpend ?? 0);
        var payments = await Db.PaymentRecords.AsNoTracking().Where(p => p.Status == "Completed").ToListAsync();
        _totalRevenue = payments.Sum(p => p.Amount ?? 0);
        _roi = _marketingExpense > 0 ? (_totalRevenue / _marketingExpense) * 100 : 0;
        _costRatio = _totalRevenue > 0 ? (_marketingExpense / _totalRevenue) * 100 : 0;
        _loaded = true;
    }
}
