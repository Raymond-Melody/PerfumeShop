@code {
    [Inject] private PerfumeShopContext Db { get; set; } = default!;
    private bool _loaded;
    private decimal _totalReceivable, _totalPayable, _liquidity;
    private int _receivableCount, _payableCount;

    protected override async Task OnInitializedAsync()
    {
        var receivables = await Db.AccountsReceivables.AsNoTracking().Where(r => r.Status == "Pending").ToListAsync();
        _totalReceivable = receivables.Sum(r => r.Amount ?? 0); _receivableCount = receivables.Count;
        var payables = await Db.AccountsPayables.AsNoTracking().Where(p => p.Status == "Pending").ToListAsync();
        _totalPayable = payables.Sum(p => p.Amount ?? 0); _payableCount = payables.Count;
        var payments = await Db.PaymentRecords.AsNoTracking().Where(p => p.Status == "Completed").ToListAsync();
        _liquidity = payments.Sum(p => p.Amount ?? 0) - _totalPayable;
        _loaded = true;
    }
}
