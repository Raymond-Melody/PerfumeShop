@code {
    [Inject] private PerfumeShopContext Db { get; set; } = default!;
    private bool _loaded;
    private decimal _totalPayment, _totalRefund, _avgPayment;
    private int _paymentCount, _refundCount;
    private List<PaymentRecord> _payments = new();

    protected override async Task OnInitializedAsync()
    {
        _payments = await Db.PaymentRecords.AsNoTracking().OrderByDescending(p => p.PaidAt).Take(200).ToListAsync();
        var refunds = await Db.RefundRecords.AsNoTracking().Where(r => r.Status == "approved").ToListAsync();
        _totalPayment = _payments.Where(p => p.Status == "Completed").Sum(p => p.Amount ?? 0); _paymentCount = _payments.Count(p => p.Status == "Completed");
        _totalRefund = refunds.Sum(r => r.Amount ?? 0); _refundCount = refunds.Count;
        _avgPayment = _paymentCount > 0 ? _totalPayment / _paymentCount : 0;
        _loaded = true;
    }
}
