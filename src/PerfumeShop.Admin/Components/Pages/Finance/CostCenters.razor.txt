@page "/admin/finance/cost-centers"
@using Microsoft.EntityFrameworkCore
@using PerfumeShop.Data.Models
@inject ISnackbar Snackbar

<PageTitle>成本中心管理</PageTitle>
<MudText Typo="Typo.h4" Class="mb-4"><i class="fas fa-building" style="color:#00bcd4;"></i> 成本中心管理</MudText>
@if (_centers is not null)
{
    @if (_editing)
    {
        <MudPaper Class="pa-6 mb-4" Elevation="0" Style="background:linear-gradient(135deg,#2d2d44,#1e1e32);border:1px solid rgba(0,188,212,0.2);border-radius:12px;max-width:600px;">
            <MudText Typo="Typo.h6" Class="mb-3" Style="color:#00bcd4;">@(_editId > 0 ? "编辑成本中心" : "新增成本中心")</MudText>
            <MudTextField @bind-Value="_formCode" Label="编码" Required="true" Variant="Variant.Outlined" Class="mb-3" />
            <MudTextField @bind-Value="_formName" Label="名称" Required="true" Variant="Variant.Outlined" Class="mb-3" />
            <MudTextField @bind-Value="_formDesc" Label="描述" Variant="Variant.Outlined" Lines="2" Class="mb-3" />
            <MudSwitch @bind-Checked="_formActive" Color="Color.Success" Label="启用" Class="mb-3" />
            <MudButton Variant="Variant.Filled" Color="Color.Primary" OnClick="SaveForm">保存</MudButton>
            <MudButton Variant="Variant.Outlined" OnClick="CancelEdit" Class="ml-2">取消</MudButton>
        </MudPaper>
    }
    <MudButton Variant="Variant.Filled" Color="Color.Primary" StartIcon="@Icons.Material.Filled.Add" OnClick="StartAdd" Class="mb-4">新增成本中心</MudButton>
    <MudTable Items="@_centers" Hover="true" Dense="true" Striped="true">
        <HeaderContent><MudTh>编码</MudTh><MudTh>名称</MudTh><MudTh>描述</MudTh><MudTh>状态</MudTh><MudTh>操作</MudTh></HeaderContent>
        <RowTemplate>
            <MudTd Style="font-weight:600;color:#00bcd4;">@context.Code</MudTd><MudTd>@context.Name</MudTd><MudTd>@context.Description</MudTd>
            <MudTd><MudChip Size="Size.Small" Color="@(context.IsActive == true ? Color.Success : Color.Default)">@(context.IsActive == true ? "启用" : "禁用")</MudChip></MudTd>
            <MudTd>
                <MudIconButton Icon="@Icons.Material.Filled.Edit" Size="Size.Small" Color="Color.Info" OnClick="@(() => StartEdit(context))" />
                <MudIconButton Icon="@Icons.Material.Filled.Delete" Size="Size.Small" Color="Color.Error" OnClick="@(() => DeleteItem(context))" />
            </MudTd>
        </RowTemplate>
    </MudTable>
}
else { <MudProgressCircular Color="Color.Primary" Indeterminate="true" Class="my-8" /> }
@code {
    [Inject] private PerfumeShopContext Db { get; set; } = default!;
    private bool _editing, _formActive; private int _editId;
    private string _formCode = "", _formName = "", _formDesc = "";
    private List<CostCenter>? _centers;

    protected override async Task OnInitializedAsync() { _centers = await Db.CostCenters.AsNoTracking().ToListAsync(); }
    private void StartAdd() { _editId = 0; _formCode = ""; _formName = ""; _formDesc = ""; _formActive = true; _editing = true; }
    private void StartEdit(CostCenter c) { _editId = c.CenterId; _formCode = c.Code ?? ""; _formName = c.Name ?? ""; _formDesc = c.Description ?? ""; _formActive = c.IsActive ?? true; _editing = true; }
    private void CancelEdit() { _editing = false; }
    private async Task SaveForm()
    {
        if (string.IsNullOrWhiteSpace(_formName)) { Snackbar.Add("请输入名称", Severity.Warning); return; }
        if (_editId > 0) { var e = await Db.CostCenters.FindAsync(_editId); if (e != null) { e.Code = _formCode; e.Name = _formName; e.Description = _formDesc; e.IsActive = _formActive; await Db.SaveChangesAsync(); } Snackbar.Add("已更新", Severity.Success); }
        else { Db.CostCenters.Add(new CostCenter { Code = _formCode, Name = _formName, Description = _formDesc, IsActive = _formActive }); await Db.SaveChangesAsync(); Snackbar.Add("已添加", Severity.Success); }
        _editing = false; _centers = await Db.CostCenters.AsNoTracking().ToListAsync();
    }
    private async Task DeleteItem(CostCenter c) { var e = await Db.CostCenters.FindAsync(c.CenterId); if (e != null) { Db.CostCenters.Remove(e); await Db.SaveChangesAsync(); Snackbar.Add("已删除", Severity.Success); _centers!.Remove(c); } }
}
