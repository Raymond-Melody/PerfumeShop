@code {
    [Inject] private PerfumeShopContext Db { get; set; } = default!;
    private bool _loaded;
    private List<Gltransaction> _items = new();
    private string _searchText = "";

    protected override async Task OnInitializedAsync() { _items = await Db.Gltransactions.AsNoTracking().OrderByDescending(g => g.TransactionDate).Take(500).ToListAsync(); _loaded = true; }
    private List<Gltransaction> GetFiltered()
    {
        var q = _items.AsEnumerable();
        if (!string.IsNullOrWhiteSpace(_searchText)) { var s = _searchText.Trim(); q = q.Where(g => (g.AccountCode != null && g.AccountCode.Contains(s)) || (g.Description != null && g.Description.Contains(s)) || (g.ReferenceNo != null && g.ReferenceNo.Contains(s))); }
        return q.ToList();
    }
}
